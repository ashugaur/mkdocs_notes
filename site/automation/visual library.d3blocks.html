    chord_simulation¶

    #%% chord simulation

import numpy as np
import pandas as pd
import random
from d3blocks import D3Blocks

sources = ['a', 'b', 'c', 'd', 'e']
targets = ['a', 'b', 'c', 'd', 'e']

# randomly generate the weights for each source and target combination
weights = []
for source in sources:
    for target in targets:
        weights.append({'source': source, 'target': target, 'weight': random.randint(1, 1000)})

# create a pandas dataframe from the data
df = pd.DataFrame(weights)

# Initialize
d3 = D3Blocks()

# Plot
d3.chord(df,
         fontsize=20,
         cmap='tab20',
         title='Chord - D3blocks',
         figsize=[600, 600],
         filepath=r'C:\my disk\edupunk\automation\visual library.d3blocks\flow\chord_simulation.html',
         )

    #%% sankey
import random
import pandas as pd
from d3blocks import D3Blocks

# Initialize
d3 = D3Blocks(chart='Sankey', frame=True)
D3Blocks.set_config()
# Import example
df = d3.import_example('energy')

# Node properties
d3.set_node_properties(df)
print(d3.node_properties)

d3.set_edge_properties(df, color='target', opacity='target')
print(d3.edge_properties)

# Show the chart
d3.show(filepath=r'C:\my disk\edupunk\automation\visual library.d3blocks\flow\sankey_sample.html')
     

    sankey_simple_simulation¶

    #%% sankey simple simulation
import random
import pandas as pd
from d3blocks import D3Blocks

# initialize
d3 = D3Blocks()

sources = ['cat', 'bat', 'rat', 'mouse', 'frog', 'bat', 'platpus']
targets = ['tiger', 'lion', 'panther', 'puma', 'jaguar']

df = pd.DataFrame()
for i in range(0,(len(sources)+len(targets))):
    df.loc[i, 'source'] = random.choice(sources)
    df.loc[i, 'target'] = random.choice(targets)
    df.loc[i, 'weight'] = random.randint(10,500)

# plot1
#note: this chart shows flow in one direction (source -> target) only, we need to break category using arrows to show inflow and outflow in both directions
d3.sankey(df,
          filepath=r'C:\my disk\edupunk\automation\visual library.d3blocks\flow\sankey_simple_simulation.html',
          figsize=(600,500),
          title='Simple simulation',
          showfig=True,
          node={"align": "justify", "width": 30, "padding": 30, "color": "currentColor"},
          margin={"top": 10, "right": 1, "bottom": 5, "left": 1},
          link={"color": "source-target", "stroke_opacity": 0.2, 'color_static': '#D3D3D3'})
